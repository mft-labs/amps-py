from .action import Action


class Endpoint(Action):
    '''The `Endpoint` class from AMPS extends the `Action` to provide additional convenience in setting up API endpoint actions. Like regular actions, endpoint actions can be performed by overriding the `action` callback exposed by the class. 

    The following attributes are available on the `Endpoint` instance.

    - msg: The msg attribute contains a python of the dictionary and the message with all of its metadata. Message data can be accessed from the msg attribute, either using the "data" key for inline data, or the "fpath" key for a path to the file containing the data. 
    - parms: The parms attribute contains all the parameters of the configured action including any extra parameters you may have specified under the "parms" key. 
    - sysparms: The sysparms attribute contains all useful system configuration parameters for use in actions. Currently, sysparms only contains the AMPS temporary directory under the "tempdir" key. 
    - provider: If use_provider is true on the configured action, the provider attribute contains the provider parameters. The provider parameters are available under the parms  object as well under the "provider" key. 
    - logger: The logger attribute exposes a Logger object for logging events back to AMPS. Any messages logged using this object during the action execution will appear in the corresponding session logs. 
    - path_params: The path_params attribute contains any parameters that were passed to the endpoint via the URL. It is also available on the `msg` attribute via the "path_params" key.

    Unlike actions, which are typically performed asynchronous via topic workflows, because Endpoints are called synchronously, if the endpoint action executes successfully, but produces an erroneous state, the action should return a successful status with the appropriate error code and response body under the "response" key.  

    #### Usage:
    ```
    from amps import Endpoint
    def my_endpoint(Endpoint):
        def action(self):
            # Perform Endpoint Logic Here
            if success:
                return Endpoint.send_resp_file("path/to/file", 200)
            else:
                return Endpoint.send_resp_data("File not found.", 404)
    ```
    '''

    def __init__(self, msgdata):
        super().__init__(msgdata)
        self.path_params = self.msg["path_params"]

    def send_resp_data(data: str, code: int):
        '''Static method for creating a dictionary with the provided inline data and status code in the "response" object for returning in the `action` callback.

        :param data: The inline data to supply in the response body.
        :param code: An integer containing the response status code to set.

        This convenience method allows for the convenient sending of inline data in the response of a request along with the specified status code.

        #### Usage:
        ```
        from amps import Endpoint
        import json
        def my_endpoint(Endpoint):
            def action(self):
                # Perform Endpoint Logic Here
                if success:
                    return Endpoint.send_resp_data(json.dumps({"id": 1, "name": "Test User"}), 200)
                else:
                    return Endpoint.send_resp_data("User not found.", 404)
        ```
        '''
        return {"status": "completed", "response": {"data": data, "code": code}}

    def send_resp_file(fpath, code):
        '''Static method for creating a dictionary with the provided file path and status code in the "response" object for returning in the `action` callback.

        :param fpath: The path to file to send in the response body.
        :param code: An integer containing the response status code to set.

        This convenience method allows for the convenient sending of a file in the response of a request along with the specified status code. 

        #### Usage:
        ```
        from amps import Endpoint
        def my_endpoint(Endpoint):
            def action(self):
                # Perform Endpoint Logic Here
                if success:
                    return Endpoint.send_resp_file("path/to/file", 200)
                else:
                    return Endpoint.send_resp_data("File not found.", 404)
        ```
        '''
        return {"status": "completed", "response": {"fpath": fpath, "code": code}}
